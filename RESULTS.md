#Résultats

##Phrases avec contexte :

> "Je suis abassie par votre moiguasse et je ne tafflerai jamais vos imnorgues."

Je : Pronoun

suis : Verb

*abassie* : Verb

par : Preposition

votre : Determiner

*moiguasse* : Noun

et : Conjunction

je : Pronoun

ne : Adverb

*tafflerai* : Verb

jamais : Adverb

vos : Determiner

*imnorgues* : Noun

> "Ce cacus abules est véritablement sylecouris."

Ce : Determiner

*cacus* : Noun

*abules* : Adjective

est : Verb

véritablement : Adverb

*sylecouris* : Adjective

##Phrases sans contexte :

> "emboulise gableutes nierant schisse clont"

emboulise : Adjective

gableutes : Noun

nierant : Preposition

schisse : Noun

clont : Adjective

> "soclatique mandasions masaticant mornacs"

soclatique : Adjective

mandasions : Noun

masaticant : Adjective

mornacs : Noun
